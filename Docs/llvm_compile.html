1. Native Compile

1.1. Clang

#!/usr/bin/bash

set -euo pipefail

THREADS="$(($(nproc --all) - 1))"

mkdir -p build_clang

pushd "build_clang" || return

cmake                                      \
     -DLLVM_ENABLE_PROJECTS="clang;lld"    \
     -DCMAKE_BUILD_TYPE=Release            \
     -DLLVM_ENABLE_RUNTIMES="compiler-rt"  \
     -DLLVM_INSTALL_UTILS=ON               \
     -DCMAKE_INSTALL_PREFIX="$INSTALL_DIR" \
     -DLLVM_ENABLE_OCAMLDOC=OFF            \
     -DLLVM_ENABLE_BINDINGS=OFF            \
     -DLLVM_BUILD_LLVM_DYLIB=ON            \
     -DLLVM_LINK_LLVM_DYLIB=ON             \
     -DLLVM_ENABLE_FFI=ON                  \
     -DLLVM_ENABLE_RTTI=ON                 \
     -DLLVM_ENABLE_ASSERTIONS=OFF          \
     -DLLVM_APPEND_VC_REV=OFF              \
     -G "Unix Makefiles"                   \
      ../llvm
make -j$THREADS

exit 0

2. Cross Compile

2.1. Clang

#!/usr/bin/bash

set -euo pipefail

THREADS="$(($(nproc --all) - 1))"

mkdir -p build_clang

pushd "build_clang" || return

cmake \
      -DLLVM_ENABLE_PROJECTS=clang \
      -DCMAKE_BUILD_TYPE=Release                                \
      -DLLVM_ENABLE_RUNTIMES="compiler-rt" \
      -DLLVM_DEFAULT_TARGET_TRIPLE="loongarch64-unknown-linux-gnu" \
      -G "Unix Makefiles" \
      ../llvm
make -j$THREADS

exit 0

2.2. Compiler-RT

Script to build Compiler-RT for loongarch64: 

#!/usr/bin/bash

set -euo pipefail

THREADS="$(($(nproc --all) - 1))"

mkdir -p build_compiler-rt
mkdir -p build_runtimes

pushd "build_compiler-rt" || return

cmake \
      -DCMAKE_BUILD_TYPE=Release                                \
      -G "Unix Makefiles" \
      -DCMAKE_C_FLAGS="--target=loongarch64-unknown-linux-gnu -march=loongarch64" \
      -DCMAKE_CXX_FLAGS="--target=loongarch64-unknown-linux-gnu -march=loongarch64" \
      -DCMAKE_C_COMPILER=/home/jlc/Codebase/llvm-project/native_build/bin/clang \
      -DCMAKE_AR=/home/jlc/Codebase/llvm-project/native_build/bin/llvm-ar \
      -DCMAKE_NM=/home/jlc/Codebase/llvm-project/native_build/bin/llvm-nm \
      -DCMAKE_RANLIB=/home/jlc/Codebase/llvm-project/native_build/bin/llvm-ranlib \
      -DCMAKE_OBJCOPY=/home/jlc/Codebase/llvm-project/native_build/bin/llvm-objcopy \
      -DCOMPILER_RT_BUILD_BUILTINS=ON \
      -DCOMPILER_RT_BUILD_LIBFUZZER=ON \
      -DCOMPILER_RT_BUILD_MEMPROF=ON \
      -DCOMPILER_RT_BUILD_PROFILE=ON \
      -DCOMPILER_RT_BUILD_SANITIZERS=ON \
      -DCOMPILER_RT_BUILD_XRAY=OFF \
      -DCOMPILER_RT_DEFAULT_TARGET_ONLY=ON \
      -DPython3_EXECUTABLE:PATH=/usr/bin/python3 \
      -DCMAKE_C_COMPILER_TARGET="loongarch64-unknown-linux-gnu" \
      -DCMAKE_ASM_COMPILER_TARGET="loongarch64-unknown-linux-gnu" \
      -DCMAKE_C_COMPILER_EXTERNAL_TOOLCHAIN=/home/jlc/cross_tools/cross-tools/ \
      -DCMAKE_SYSROOT="/home/jlc/clfs_workspace/loongarch64_root" \
      ../compiler-rt
make -j$THREADS

exit 0
