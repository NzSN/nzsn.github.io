  • Constexpr string 

    template<std::size_t N>
struct ConstStr {

  constexpr ConstStr() = default;

  template<std::size_t M>
  constexpr ConstStr(const char (&other)[M]) {
    if (N < M) {
      throw std::runtime_error("...");
    }

    reset();
    for (unsigned i = 0; i < M; ++i) {
      data[i] = other[i];
    }
  }

  template<std::size_t M>
  constexpr ConstStr& operator=(const char (&other)[M]) {
    if (N < M) {
      return *this;
    }

    reset();
    for (unsigned i = 0; i < M; ++i) {
      data[i] = other[i];
    }

    return *this;
  }

  constexpr void reset() {
    for (unsigned i = 0; i < N; ++i) {
      data[i] = 0;
    }
  }

  char data[N+1];
};

2. Compile-Time reflection

#define Stringify(...) #__VA_ARGS__
constexpr const unsigned MAXIMUM_CLASSNAME = 256;
constexpr unsigned NUM_OF_REFLECTS = 30;

template<typename T, typename... ARGS>
struct Reflection {
  static constexpr char const* Type{ "Unknown" };
  static constexpr T (*Construct)(ARGS...) = nullptr;
};

#define REGISTER_TYPE_TO_REFLECTION(T, CTOR, ...)                 \
  template<>                                                      \
  struct Utility::Reflection<T> {                                 \
    static constexpr char const* Type{ Stringify(T) };            \
    static constexpr T (*Construct)(__VA_ARGS__) = CTOR;          \
  }


 struct Base {
  int value() const { return 0; }
};

struct Derived: public Base {
  int value() const { return 1; }
};

constexpr Base ctor() {
  return Base{};
}

constexpr Derived ctor_D() {
  return Derived{};
}

REGISTER_TYPE_TO_REFLECTION(Base, ctor);
REGISTER_TYPE_TO_REFLECTION(Derived, ctor_D);

3. How to get stack trace at runtime ?

3.1. Via glibc

#include <execinfo.h>
#include <malloc.h>
#include <unistd.h>
#include <string.h>
#include <errno.h>

static void full_write(int fd, const char *buf, size_t len)
{
        while (len > 0) {
                ssize_t ret = write(fd, buf, len);

                if ((ret == -1) && (errno != EINTR))
                        break;

                buf += (size_t) ret;
                len -= (size_t) ret;
        }
}
void print_backtrace(void)
{
        static const char start[] = "BACKTRACE ------------\n";
        static const char end[] = "----------------------\n";

        void *bt[1024];
        int bt_size;
        char **bt_syms;
        int i;

        bt_size = backtrace(bt, 1024);
        bt_syms = backtrace_symbols(bt, bt_size);
        full_write(STDERR_FILENO, start, strlen(start));
        for (i = 1; i < bt_size; i++) {
                size_t len = strlen(bt_syms[i]);
                full_write(STDERR_FILENO, bt_syms[i], len);
                full_write(STDERR_FILENO, "\n", 1);
        }
        full_write(STDERR_FILENO, end, strlen(end));
    free(bt_syms);
}

void f() { print_backtrace(); }

int main() {
  f();
  return 0;
}
