  • Virtual move constructor

    Move constructor of this class is in ill-formed, due to it declared with a const parameter which implies that it’s parameter cannot be modified but move semantic require to modify it. 

    template<Base::GPTMeta T>
class CondExpr {
public:
  // Default constructor will construct an
  // empty CondExpr.
  CondExpr(): condition_{nullptr} {}

  CondExpr(const CondExpr& ce) = delete;

  // FIXME: This signature is wrong, cause the object
  //        be moved cannot be const.
  CondExpr(const CondExpr&& ce):
    condition_{std::move(ce.condition_)} {}

  CondExpr(std::unique_ptr<Expression::Expr<T>>&& condition):
    condition_{std::move(condition)} {}

  bool isEmpty() const {
    return condition_.get() == nullptr;
  }

  bool operator()(Environment<T>* env);

private:
  std::unique_ptr<Expression::Expr<T>> condition_;
};

  • Constexpr string 

    template<std::size_t N>
struct ConstStr {

  constexpr ConstStr() = default;

  template<std::size_t M>
  constexpr ConstStr(const char (&other)[M]) {
    if (N < M) {
      throw std::runtime_error("...");
    }

    reset();
    for (unsigned i = 0; i < M; ++i) {
      data[i] = other[i];
    }
  }

  template<std::size_t M>
  constexpr ConstStr& operator=(const char (&other)[M]) {
    if (N < M) {
      return *this;
    }

    reset();
    for (unsigned i = 0; i < M; ++i) {
      data[i] = other[i];
    }

    return *this;
  }

  constexpr void reset() {
    for (unsigned i = 0; i < N; ++i) {
      data[i] = 0;
    }
  }

  char data[N+1];
};

2. Compile-Time reflection

#define Stringify(...) #__VA_ARGS__
constexpr const unsigned MAXIMUM_CLASSNAME = 256;
constexpr unsigned NUM_OF_REFLECTS = 30;

template<typename T, typename... ARGS>
struct Reflection {
  static constexpr char const* Type{ "Unknown" };
  static constexpr T (*Construct)(ARGS...) = nullptr;
};

#define REGISTER_TYPE_TO_REFLECTION(T, CTOR, ...)                 \
  template<>                                                      \
  struct Utility::Reflection<T> {                                 \
    static constexpr char const* Type{ Stringify(T) };            \
    static constexpr T (*Construct)(__VA_ARGS__) = CTOR;          \
  }


 struct Base {
  int value() const { return 0; }
};

struct Derived: public Base {
  int value() const { return 1; }
};

constexpr Base ctor() {
  return Base{};
}

constexpr Derived ctor_D() {
  return Derived{};
}

REGISTER_TYPE_TO_REFLECTION(Base, ctor);
REGISTER_TYPE_TO_REFLECTION(Derived, ctor_D);
